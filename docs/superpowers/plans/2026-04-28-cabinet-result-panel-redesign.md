# Cabinet Result Panel Redesign Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Reorganize the cabinet unit result panel so doors/hinges, drawers, and internal components each appear in their own clearly-separated section, with bold subtotals per section and a corrected cost breakdown.

**Architecture:** Add `group` / `parentDoorId` tags to `PanelResult` and `HardwareResult` in the type definitions, set them in the calculation engine, then use them in the display component to filter rows into the correct sections. Four new summary fields (`bodyBoardOnlyCost`, `doorTotalCost`, `drawerCost`, `extraHardwareCost`) drive the updated cost breakdown.

**Tech Stack:** TypeScript (strict), Next.js 15 App Router, Tailwind CSS, React

"use client";

import { Fragment } from "react";
import type { CabinetUnitResult, DoorResult, HardwareResult, PanelResult } from "@/types";
import { formatCurrency, formatNumber } from "@/lib/utils";

import { Card, CardContent } from "@/components/ui/card";
import { Badge } from "@/components/ui/badge";

interface Props {
  result: CabinetUnitResult;

type BoardRow = PanelResult | DoorResult;

function unitPrice(row: BoardRow): number {
  return "unitCost" in row ? row.unitCost : row.materialRef?.pricePerUnit ?? 0;
}

function isAutoGenerated(row: BoardRow): boolean {
  return "isAutoGenerated" in row && row.isAutoGenerated;
}

function rowProcessCost(row: BoardRow): number {
  return "processes" in row
    ? row.processes.reduce((acc, process) => acc + (process.includedInSubtotal ? process.cost : 0), 0)
    : 0;
}

function rowBaseSubtotal(row: BoardRow): number {
  return row.subtotal - rowProcessCost(row);
}

function processLabel(label: string): string {
  return `加工-${label.replace(": ", ":")}`;
}

function HardwareSubRows({ hardware }: { hardware: HardwareResult[] }) {
  return (
    <>
      {hardware.map((hw) => (
        <tr key={hw.id} className="border-b border-muted/30 bg-muted/10">
          <td className="py-1 pl-4 text-muted-foreground">
            {hw.name}
            {hw.description && (
              <span className="block text-[10px]">{hw.description}</span>
            )}
          </td>
          <td className="py-1 text-right text-muted-foreground">-</td>
          <td className="py-1 text-right text-muted-foreground">{hw.quantity}</td>

          <td className="py-1 text-right text-muted-foreground">
            {hw.materialRef ? (
              <>
                {formatCurrency(hw.unitCost)}
                <span className="text-muted-foreground">/{hw.materialRef.unit}</span>
              </>
            ) : hw.unitCost > 0 ? (
              formatCurrency(hw.unitCost)
            ) : (
              <span className="text-muted-foreground">-</span>
            )}
          </td>
          <td className="max-w-[160px] truncate py-1 text-right text-muted-foreground">
            {hw.materialRef?.materialName ?? (hw.unitCost > 0 ? "加工費" : <span className="text-orange-500">未選材料</span>)}
          </td>
          <td className="py-1 text-right font-medium">{formatCurrency(hw.subtotal)}</td>

      ))}

}

function BoardTable({

      {title && (

            <th className="py-1 text-right font-medium">數量</th>
            <th className="py-1 text-right font-medium">才數</th>
            <th className="py-1 text-right font-medium">單價</th>
            <th className="py-1 text-right font-medium">材料</th>
            <th className="py-1 text-right font-medium">小計</th>
          </tr>
        </thead>
        <tbody>
          {rows.map((row) => (
            <Fragment key={row.id}>
              <tr className="border-b border-muted/30 hover:bg-muted/20">
                <td className="py-1">
                  {row.name}
                  {isAutoGenerated(row) && (
                    <Badge variant="outline" className="ml-1 px-1 py-0 text-[10px]">自動</Badge>
                  )}
                  {"type" in row && (
                    <Badge variant="secondary" className="ml-1 px-1 py-0 text-[10px]">
                      {row.type === "HINGED" ? "鉸鏈門" : "滑門"}
                    </Badge>
                  )}
                </td>
                <td className="py-1 text-right text-muted-foreground">{row.widthCm} x {row.heightCm}</td>
                <td className="py-1 text-right">{row.quantity}</td>
                <td className="py-1 text-right">
                  <AreaDisplay area={row.totalArea} />
                  {row.billableTotalArea.cai > row.totalArea.cai && (
                    <span className="block text-[10px] text-orange-600">
                      計價 <AreaDisplay area={row.billableTotalArea} />
                    </span>
                  )}
                </td>
                <td className="py-1 text-right">
                  {row.materialRef ? (
                    <>
                      {formatCurrency(unitPrice(row))}
                      <span className="text-muted-foreground">/{row.materialRef.unit}</span>
                    </>
                  ) : (
                    <span className="text-muted-foreground">-</span>
                  )}
                </td>
                <td className="max-w-[160px] truncate py-1 text-right text-muted-foreground">
                  {row.materialRef?.materialName ?? <span className="text-orange-500">未選材料</span>}
                </td>
                <td className="py-1 text-right font-medium">{formatCurrency(rowBaseSubtotal(row))}</td>
              </tr>
              {"processes" in row &&
                row.processes.map((process) => (
                  <tr key={process.id} className="border-b border-muted/30 bg-muted/10">
                    <td className="py-1 pl-4 text-muted-foreground">{processLabel(process.label)}</td>
                    <td className="py-1 text-right text-muted-foreground">-</td>
                    <td className="py-1 text-right text-muted-foreground">1</td>
                    <td className="py-1 text-right text-muted-foreground">-</td>
                    <td className="py-1 text-right text-muted-foreground">-</td>
                    <td className="py-1 text-right text-muted-foreground">後加工</td>
                    <td className="py-1 text-right font-medium">{formatCurrency(process.cost)}</td>
                  </tr>
                ))}
              {"type" in row && doorHardware?.has(row.id) && (
                <HardwareSubRows hardware={doorHardware.get(row.id)!} />
              )}
            </Fragment>
          ))}
        </tbody>
        {subtotalLabel !== undefined && subtotalAmount !== undefined && (
          <tfoot>
            <tr className="border-t-2 border-muted/50">
              <td
                colSpan={6}
                className="py-1.5 text-right text-xs font-semibold text-muted-foreground"
              >
                {subtotalLabel}
              </td>
              <td className="py-1.5 text-right text-sm font-bold">
                {formatCurrency(subtotalAmount)}
              </td>
            </tr>
          </tfoot>
        )}
      </table>
    </div>
  );
}

function HardwareTable({
  title,
  rows,
}: {
  title?: string;

  return (

                )}
              </td>
              <td className="py-1 text-right">{hw.quantity}</td>

                  <>
                    {formatCurrency(hw.unitCost)}
                    <span className="text-muted-foreground">/{hw.materialRef.unit}</span>

                ) : hw.unitCost > 0 ? (

            </tr>
          ))}
        </tbody>
      </table>
    </div>
  );
}

export function CabinetResultPanel({ result }: Props) {
  const { summary } = result;

  const internalOnly = result.internalParts.filter((p) => p.group === "internal");
  const drawerPanels = result.internalParts.filter((p) => p.group === "drawer_panel");
  const internalOnlyCost = internalOnly.reduce((acc, p) => acc + p.subtotal, 0);

  const doorHardwareMap = new Map<string, HardwareResult[]>();
  const drawerHardware: HardwareResult[] = [];
  const extraHardware: HardwareResult[] = [];

  for (const hw of result.hardware) {
    if (hw.group === "door" && hw.parentDoorId) {
      const existing = doorHardwareMap.get(hw.parentDoorId) ?? [];
      existing.push(hw);
      doorHardwareMap.set(hw.parentDoorId, existing);
    } else if (hw.group === "drawer") {
      drawerHardware.push(hw);
    } else if (hw.group === "extra") {
      extraHardware.push(hw);
    }
  }

  const hasDrawerSection = drawerPanels.length > 0 || drawerHardware.length > 0;

  return (
    <div className="space-y-4">
      <div className="grid grid-cols-2 gap-3">
        <Card className="bg-muted/30">
          <CardContent className="p-3">
            <p className="text-xs text-muted-foreground">總板材面積</p>
            <p className="text-lg font-bold">{formatNumber(summary.totalAreaCai, 2)} 才</p>
            <p className="text-xs text-muted-foreground">{formatNumber(summary.totalAreaM2, 3)} m2</p>
          </CardContent>
        </Card>
        <Card className="bg-primary/5">
          <CardContent className="p-3">
            <p className="text-xs text-muted-foreground">估算總價</p>
            <p className="text-lg font-bold text-primary">{formatCurrency(summary.totalCost)}</p>
          </CardContent>
        </Card>
      </div>

      <div className="space-y-4">
        <BoardTable title="桶身板件" rows={result.panels} />

        <BoardTable
          title="門片"
          rows={result.doors}
          doorHardware={doorHardwareMap}
          subtotalLabel={result.doors.length > 0 ? "門片合計" : undefined}
          subtotalAmount={result.doors.length > 0 ? summary.doorTotalCost : undefined}
        />

        {hasDrawerSection && (
          <div className="space-y-2">
            <h4 className="text-sm font-semibold uppercase tracking-wide text-muted-foreground">
              抽屜
            </h4>
            <BoardTable title="" rows={drawerPanels} />
            <HardwareTable rows={drawerHardware} />
            <div className="flex justify-end border-t border-muted/50 pt-1.5">
              <span className="mr-4 text-xs font-semibold text-muted-foreground">抽屜合計</span>
              <span className="text-sm font-bold">{formatCurrency(summary.drawerCost)}</span>
            </div>
          </div>
        )}

        {internalOnly.length > 0 && (
          <BoardTable
            title="內部構件（中隔板 / 層板）"
            rows={internalOnly}
            subtotalLabel="內部構件合計"
            subtotalAmount={internalOnlyCost}
          />
        )}

        <HardwareTable title="五金/另料" rows={extraHardware} />

        {result.accessories.length > 0 && (
          <div className="overflow-x-auto">
            <h4 className="mb-2 text-sm font-semibold uppercase tracking-wide text-muted-foreground">
              配件
            </h4>
            <table className="w-full text-xs">
              <tbody>
                {result.accessories.map((accessory) => (
                  <tr key={accessory.id} className="border-b border-muted/30 hover:bg-muted/20">
                    <td className="py-1">{accessory.name}</td>
                    <td className="py-1 text-right text-muted-foreground">
                      {accessory.widthCm} x {accessory.heightCm} cm
                    </td>
                    <td className="py-1 text-right">{accessory.quantity}</td>
                    <td className="py-1 text-right font-medium">{formatCurrency(accessory.subtotal)}</td>
                  </tr>
                ))}
              </tbody>
            </table>
          </div>
        )}
      </div>

      <Separator />
      <div className="space-y-1 text-sm">
        <div className="flex justify-between text-muted-foreground">
          <span>桶身板材</span>
          <span>{formatCurrency(summary.bodyBoardOnlyCost)}</span>
        </div>
        {summary.boardBackingCost > 0 && (
          <div className="flex justify-between text-muted-foreground">
            <span>背板</span>
            <span>{formatCurrency(summary.boardBackingCost)}</span>
          </div>
        )}
        {summary.doorTotalCost > 0 && (
          <div className="flex justify-between text-muted-foreground">
            <span>門片</span>
            <span>{formatCurrency(summary.doorTotalCost)}</span>
          </div>
        )}
        {summary.drawerCost > 0 && (
          <div className="flex justify-between text-muted-foreground">
            <span>抽屜</span>
            <span>{formatCurrency(summary.drawerCost)}</span>
          </div>
        )}
        {summary.extraHardwareCost > 0 && (
          <div className="flex justify-between text-muted-foreground">
            <span>五金</span>
            <span>{formatCurrency(summary.extraHardwareCost)}</span>
          </div>
        )}
        {summary.addonsCost > 0 && (
          <div className="flex justify-between text-muted-foreground">
            <span>加工費</span>
            <span>{formatCurrency(summary.addonsCost)}</span>
          </div>
        )}
        <Separator />
        <div className="flex justify-between text-base font-bold">
          <span>總計</span>
          <span className="text-primary">{formatCurrency(summary.totalCost)}</span>
        </div>
      </div>
    </div>
  );
}
